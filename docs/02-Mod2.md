<script>

let dragged;

document.querySelectorAll('.leaf').forEach(el=>{
el.setAttribute("draggable","true");

el.addEventListener("dragstart",function(e){
dragged = e.target;
});
});

document.querySelectorAll('.stem').forEach(stem=>{

stem.addEventListener("dragover",function(e){
e.preventDefault();
});

stem.addEventListener("drop",function(e){
e.preventDefault();
stem.appendChild(dragged);
});

});

function showHint(){
document.getElementById("hint").style.display="block";
}

function checkAnswer(){

const correct = {
"2.3":["7"],
"2.5":["3"],
"2.6":["1","3","5","7","8","9"],
"2.7":["2","3","8","9"],
"2.8":["4","5","5"]
};

let correctStem = 0;
let ordered = 0;

document.querySelectorAll(".stem").forEach(stem=>{

let label = stem.dataset.stem;
let leaves = Array.from(stem.querySelectorAll(".leaf")).map(x=>x.textContent);

let sorted = [...leaves].sort();

if(JSON.stringify(sorted) === JSON.stringify(correct[label])){
correctStem++;

if(JSON.stringify(leaves) === JSON.stringify(sorted)){
ordered++;
}

}

});

if(correctStem === 5 && ordered === 5){
document.getElementById("result").innerHTML="✔ Correct! Stems and leaves are correct and ordered.";
}
else if(correctStem === 5){
document.getElementById("result").innerHTML="⚠ Half correct: stems are correct but leaves are not ordered.";
}
else{
document.getElementById("result").innerHTML="✘ Some stems are incorrect. Try again.";
}

}

function resetActivity(){

const pool = document.getElementById("leaf-pool");

document.querySelectorAll(".leaf").forEach(leaf=>{
pool.appendChild(leaf);
});

document.getElementById("result").innerHTML="";
document.getElementById("hint").style.display="none";

}

</script>
